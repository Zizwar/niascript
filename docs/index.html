NiaScript

Intent-based programming language for the AI era. Transform natural language into executable code. 

import { nia } from 'niascript';

// Simple cryptocurrency price query
const btcPrice = await nia`Bitcoin price`;
console.log(btcPrice); // "BTC price: $45,000 USD"

// Investment calculation
const investment = await nia`If I invest $1000 at 8% annual return, 
what will I have after 5 years?`;
console.log(investment); // "Investment of $1000 at 8% annually for 5 years = $1469.33"

Trading Bot

class TradingBot {
  constructor() {
    this.nia = new NiaScript();
  }
  
  async analyzeMarket(query) {
    return await this.nia.processIntent(query);
  }
  
  async executeTrade(command) {
    return await this.nia.ask(command);
  }
}

const bot = new TradingBot();
const analysis = await bot.analyzeMarket(
  "What's the trend for Bitcoin in the last week?"
);

Custom Providers

// Register custom provider
nia.registerProvider('financial', 'alpha_vantage', {
  name: 'Alpha Vantage API',
  baseURL: 'https://www.alphavantage.co/query',
  apiKey: process.env.ALPHA_VANTAGE_KEY,
  rateLimits: { requests: 5, window: 60000 },
  cost: 0,
  reliability: 0.95
});

// Custom error handling
nia.onError(async (error, context) => {
  if (error.type === 'API_FAILURE') {
    return await context.tryFallback();
  }
  throw error;
});

Memory Management

// Save user preferences
await nia.remember("User prefers cryptocurrency over stocks");
await nia.remember("Risk tolerance: moderate");

// Ask questions with context
const advice = await nia.ask(
  "What should I invest in next?"
);

// Clear specific memories
await nia.forget("old investment preferences");

// Get clarification
const choice = await nia.clarify(
  "Apple could refer to:", 
  ["stock", "fruit", "company"]
);

API Reference

Complete reference for all NiaScript methods and capabilities

Core Methods

processIntent(query, values)

Process natural language query into executable actions

ask(query, options)

Ask a direct question and get intelligent response

clarify(question, choices)

Request clarification when intent is ambiguous

Memory & Providers

remember(fact)

Save information to long-term memory

forget(pattern)

Remove specific information from memory

registerProvider(category, name, config)

Register new data provider for external APIs

Quick Start

Get up and running with NiaScript in minutes

Installation 

# Install via npm
npm install niascript

# Install via yarn
yarn add niascript

# Install globally
npm install -g niascript

Basic Setup 

import { nia, NiaScript } from 'niascript';

// Simple usage with template literals
const price = await nia`Bitcoin price`;
console.log(price);

// Advanced usage with full instance
const niaInstance = new NiaScript();
const result = await niaInstance.processIntent("What's the weather like?");
console.log(result);

Environment Configuration 

# Required API keys
OPENAI_API_KEY=your_openai_key_here
BINANCE_API_KEY=your_binance_key_here
ALPHA_VANTAGE_KEY=your_alphavantage_key_here

# Optional settings
NIA_LOG_LEVEL=info
NIA_MEMORY_SIZE=1000
NIA_CACHE_TTL=300
